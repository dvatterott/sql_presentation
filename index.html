Group By Statement

Create a list of the 10 people that watched the most "Star Trek: Voyager" episode ones.


							 SELECT
							     user_id,
							     COUNT(episode_number) AS e1_count
							 FROM
							     fake_table
							 WHERE
							     series_name = "Star Trek: Voyager"
							     AND
							     episode_number = 1.0
							 GROUP BY
							     user_id
							 ORDER BY
							     COUNT(episode_number)
							 LIMIT 10

Group By Statement: The Return

Create a list of then people that watched the all "Star Trek: Voyager" episode ones.


							SELECT
							    user_id,
							    COUNT(episode_number) AS e1_count
							FROM
							    fake_table
							WHERE
							    series_name = "Star Trek: Voyager"
							    AND
							    episode_number = 1.0
							GROUP BY
							    user_id
							HAVING
							    COUNT(episode_number) = 7

Join Qusetion

Create a list of the ten people who watched the most non-male-lead episodes.


						SELECT
						    ft.user_id,
						    COUNT(ft.episode_number) AS e1_count
						FROM
						    fake_table AS ft
						    LEFT JOIN fake_table_2 AS ft2
						      ON ft.seies_name = ft2.series_name
						WHERE
						    ft2.male_lead = False
						GROUP BY
						    ft.user_id
						ORDER BY
						    COUNT(ft.episode_number)
						LIMIT 10

Sub-Queries

Create a list of the ten people who watched the most non-male-lead shows.


						SELECT
						    ft.user_id,
						    COUNT(ft.series_name) AS series_count
						FROM
						    (SELECT
						         DISTINCT(user_id, series_name)
						     FROM
						         fake_table
						    ) AS ft
						    LEFT JOIN fake_table_2 AS ft2
						      ON ft.seies_name = ft2.series_name
						WHERE
						    ft2.male_lead = False
						GROUP BY
						    ft.user_id
						ORDER BY
						    COUNT(ft.series_name)
						LIMIT 10

SQL makes my head hurt

Sub-Queries with a temp table

Create a list of the ten people who watched the most non-male-lead shows.


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id, series_name)
						    FROM
						        fake_table
						)
						SELECT
						    ft.user_id,
						    COUNT(ft.series_name) AS series_count
						FROM
						    temp_table AS ft
						    LEFT JOIN fake_table_2 AS ft2
						      ON ft.seies_name = ft2.series_name
						WHERE
						    ft2.male_lead = False
						GROUP BY
						    ft.user_id
						ORDER BY
						    COUNT(ft.series_name)
						LIMIT 10

Much better!

Exist Statements

Create a list of ten people who watched Star Trek: Voyager and Star Trek: DS9.


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id) AS user_id
						    FROM
						        fake_table
						    WHERE
						        series_name = 'Star Trek: Voyager'
						)
						SELECT
						    DISTINCT(user_id) AS user_id
						FROM
						    fake_table AS ft
						WHERE
						    series_name = 'Star Trek: Voyager'
						    AND
						    EXISTS (
							SELECT
					              1
						        FROM
						            temp_table
						        WHERE
						            temp_table.user_id = ft.user_id)

Case Statements

Count whether more episodes are watched on the weekends or weekdays.


					SELECT
          CASE
              WHEN extract(isodow from date date) > 5 THEN 'weekend'
              ELSE 'weekday'
          END AS weekend_or_not
          COUNT(*)
					FROM
					    fake_table AS ft
					GROUP BY
          CASE
              WHEN extract(isodow from date date) > 5 THEN 'weekend'
              ELSE 'weekday'
					     END

Window Functions

Create a list of users and the first series they wathced.


					WITH temp_table AS (
	        SELECT
					        user_id,
					        series_name,
					        rank() OVER (PARTITION BY user_id ORDER BY date ASC)
					    FROM
					        fake_table
					)
					SELECT
					    user_id,
					    series_name
					FROM
					    temp_table
					WHERE
					    rank = 1

Optimization/Run Order

  1. FROM
  2. ON
  3. OUTER
  4. WHERE
  5. GROUP BY
  6. HAVING
  7. SELECT
  8. DISTINCT
  9. ORDER BY
  10. LIMIT/TOP

Temp tables are faster when many tables. Joins are faster when using fewer tables.
