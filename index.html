Group By Statement

Create a list of the 10 people that watched the most "Star Trek: Voyager" episode ones.


							 SELECT
							     user_id,
							     COUNT(episode_id) AS e1_count
							 FROM
							     viewing_table
							 WHERE
							     series_name = "Star Trek: Voyager"
							     AND
							     episode_id = 1.0
							 GROUP BY
							     user_id
							 ORDER BY
							     e1_count DESC
							 LIMIT 10

Group By Statement: The Return

Create a list of the people that watched the all "Star Trek: Voyager" episode ones.


							SELECT
							    user_id,
							    COUNT(episode_id) AS e1_count
							FROM
							    viewing_table
							WHERE
							    series_name = "Star Trek: Voyager"
							    AND
							    episode_id = 1.0
							GROUP BY
							    user_id
							HAVING
							    COUNT(episode_id) = 7

Join Question

Create a list of the ten people who watched the most epsidoes from a series with a non-male-lead.


						SELECT
						    vt.user_id,
						    COUNT(DISTINCT(ft.episode_id)) AS e1_count
						FROM
						    viewing_table AS vt
						    INNER JOIN series_table AS st
						      ON vt.series_name = st.series_name
						WHERE
						    st.male_lead = 0
						GROUP BY
						    vt.user_id
						ORDER BY
						    e1_count
						LIMIT 10

Sub-Queries

Create a list of the ten people who watched the most non-male-lead shows.


						SELECT
						    vt.user_id,
						    COUNT(ft.series_name) AS series_count
						FROM
						    (SELECT
						         DISTINCT(user_id, series_name)
						     FROM
						         viewing_table
						    ) AS vt
						    INNER JOIN series_table AS st
						      ON vt.seies_name = st.series_name
						WHERE
						    st.male_lead = False
						GROUP BY
						    vt.user_id
						ORDER BY
						    series_count
						LIMIT 10

SQL makes my head hurt

Sub-Queries with a temp table

Create a list of the ten people who watched the most non-male-lead shows.


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id, series_name)
						    FROM
						        viewing_table
						)
						SELECT
						    vt.user_id,
						    COUNT(ft.series_name) AS series_count
						FROM
						    temp_table AS st
						    INNER JOIN series_table AS st
						      ON vt.seies_name = st.series_name
						WHERE
						    st.male_lead = 0
						GROUP BY
						    vt.user_id
						ORDER BY
						    series_count
						LIMIT 10

Much better!

Exist Statements

Create a list of people who watched Star Trek: Voyager and Star Trek: DS9.


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id) AS user_id
						    FROM
						        viewing_table
						    WHERE
						        series_name = 'Star Trek: Voyager'
						)
						SELECT
						    DISTINCT(user_id) AS user_id
						FROM
						    viewing_table
						WHERE
						    series_name = 'Star Trek: DS9'
						    AND
						    EXISTS (
							SELECT
					              1
						        FROM
						            temp_table
						        WHERE
						            temp_table.user_id = vt.user_id)

In Statements

Create a list of people who watched Star Trek: Voyager and Star Trek: DS9.


						WITH temp_table AS (

						        DISTINCT(user_id) AS user_id
						    FROM
						        viewing_table
						    WHERE
						        series_name = 'Star Trek: Voyager'
						)
						SELECT
						    DISTINCT(user_id) AS user_id
						FROM
						    viewing_table
						WHERE
						    series_name = 'Star Trek: DS9'
						    AND
						    user_id IN (SELECT user_id FROM temp_table)

Case Statements

Count whether more episodes are watched on the weekends or weekdays.


					SELECT
          CASE
              WHEN extract('isodow' from date) > 5 THEN 'weekend'
              ELSE 'weekday'
          END AS weekend_or_not,
          COUNT(*)
					FROM
					    viewing_table AS vt
					GROUP BY
					    weekend_or_not

Window Functions

Create a list of users and the first series they watched.


					WITH temp_table AS (
	        SELECT
					        user_id,
					        series_name,
					        rank() OVER (PARTITION BY user_id ORDER BY date ASC)
					    FROM
					        viewing_table
					)
					SELECT
					    user_id,
					    series_name
					FROM
					    temp_table
					WHERE
					    rank = 1

Optimization/Run Order

  1. FROM
  2. ON
  3. OUTER
  4. WHERE
  5. GROUP BY
  6. HAVING
  7. SELECT
  8. DISTINCT
  9. ORDER BY
  10. LIMIT/TOP

Temp tables are faster when many tables. Joins are faster when using fewer tables.
