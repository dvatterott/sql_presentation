Where statement

Create a list of people bought a "Hammer".


								SELECT
								    DISTINCT(user_id)
								FROM
								    buy_table
								WHERE
								    tool_category = "Hammer"

Where statement: Reloaded

Create a list of people bought a "screw" or "Screw driver"?


							 SELECT
							     DISTINCT(user_id)
							 FROM
							     buy_table
							 WHERE
							     tool_category ILIKE "screw%"

Group By Statement

Create a list of the 10 people that bought the most "Hammers" where the id number is 1.


							 SELECT
							     user_id,
							     COUNT(tool_id) AS buy_count
							 FROM
							     buy_table
							 WHERE
							     tool_category = "Hammer"
							     AND
							     tool_id = 1.0
							 GROUP BY
							     user_id
							 ORDER BY
							     buy_count DESC
							 LIMIT 10

Group By Statement: The Return

Create a list of the people that bought more than 7 "screws".


							SELECT
							    user_id,
							    COUNT(tool_id) AS buy_count
							FROM
							    buy_table
							WHERE
							    tool_category = "screws"
							GROUP BY
							    user_id
							HAVING
							    COUNT(tool_id) > 7

Note: Can't use select alias in where/having :/

tool_category hand_tool mechanical

Join Question

Create a list of the ten people who bought the most "hand tools".


						SELECT
						    bt.user_id,
						    COUNT(DISTINCT(bt.tool_id)) AS buy_count
						FROM
						    buy_table AS bt
						    INNER JOIN tool_table AS tt
						      ON bt.tool_category = tt.tool_category
						WHERE
						    tt.hand_tool = 1
						GROUP BY
						    bt.user_id
						ORDER BY
						    buy_count
						LIMIT 10

Sub-Queries

Create a list of the ten people who bought an item from the most distinct "hand tool" categories.


						SELECT
						    bt.user_id,
						    COUNT(tt.tool_category) AS tool_count
						FROM
						    (SELECT
						         DISTINCT(user_id, tool_category)
						     FROM
						         buy_table
						    ) AS bt
						    INNER JOIN tool_table AS tt
						      ON bt.seies_name = tt.tool_category
						WHERE
						    tt.hand_tool = 1
						GROUP BY
						    bt.user_id
						ORDER BY
						    tool_count
						LIMIT 10

SQL makes my head hurt

Sub-Queries with a temp table

Create a list of the ten people who bought an item from the most distinct "hand tool" categories.


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id, tool_category)
						    FROM
						        buy_table
						)
						SELECT
						    tt.user_id,
						    COUNT(tt.tool_category) AS tool_count
						FROM
						    temp_table AS tt
						    INNER JOIN tool_table AS tt3
						      ON tt.seies_name = tt2.tool_category
						WHERE
						    tt2.hand_tool = 0
						GROUP BY
						    tt.user_id
						ORDER BY
						    tool_count
						LIMIT 10

Much better!

Exist Statements

Create a list of people who bought a "Wrench" and a "Hammer".


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id) AS user_id
						    FROM
						        buy_table
						    WHERE
						        tool_category = 'Hammer'
						)
						SELECT
						    DISTINCT(user_id) AS user_id
						FROM
						    buy_table AS bt
						WHERE
						    tool_category = 'Wrench'
						    AND
						    EXISTS (
							SELECT
					              1
						        FROM
						            temp_table
						        WHERE
						            temp_table.user_id = bt.user_id)

In Statements

Create a list of people who bought a "Wrench" and a "Hammer".


						WITH temp_table AS (
						    SELECT
						        DISTINCT(user_id) AS user_id
						    FROM
						        buy_table
						    WHERE
						        tool_category = 'Hammer'
						)
						SELECT
						    DISTINCT(user_id) AS user_id
						FROM
						    buy_table
						WHERE
						    tool_category = 'Wrench'
						    AND
						    user_id IN (SELECT user_id FROM temp_table)

Case Statements

Count whether more items are bought on the weekends or weekdays.


					SELECT
          CASE
              WHEN extract('isodow' from buy_date) > 5 THEN 'weekend'
              ELSE 'weekday'
          END AS weekend_or_not,
          COUNT(*)
					FROM
					    buy_table AS vt
					GROUP BY
					    weekend_or_not

Window Functions

Create a list of users and the first item they bought.


					WITH temp_table AS (
	        SELECT
					        user_id,
					        tool_category,
					        rank() OVER (PARTITION BY user_id ORDER BY buy_date ASC)
					    FROM
					        buy_table
					)
					SELECT
					    user_id,
					    tool_category
					FROM
					    temp_table
					WHERE
					    rank = 1

Optimization/Run Order

  1. FROM
  2. ON
  3. OUTER
  4. WHERE
  5. GROUP BY
  6. HAVING
  7. SELECT
  8. DISTINCT
  9. ORDER BY
  10. LIMIT/TOP

Temp tables are faster when many tables. Joins are faster when using fewer tables.
